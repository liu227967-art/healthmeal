# PicEdit 图片处理网站实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个包含抠图（BiRefNet-general）、高清化、缩图功能的全栈图片处理网站。

**Architecture:** 后端使用 FastAPI 提供三个图片处理 REST API，BiRefNet-general 模型在本地运行推理；前端使用 React + Vite + Tailwind 构建响应式单页应用，通过 fetch 与后端通信，支持拖拽上传、功能切换、参数配置和结果下载。

**Tech Stack:** Python 3.10+, FastAPI, PyTorch, Transformers (BiRefNet), Pillow; React 18, Vite, Tailwind CSS 3, lucide-react

```python
# picedit/backend/services/image_service.py
import io
from PIL import Image, ImageFilter


def bytes_to_pil(data: bytes) -> Image.Image:
    return Image.open(io.BytesIO(data)).convert("RGBA")


def pil_to_png_bytes(img: Image.Image) -> bytes:
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()


def upscale_image(img: Image.Image, scale: float = 2.0) -> Image.Image:
    new_w = int(img.width * scale)
    new_h = int(img.height * scale)
    upscaled = img.resize((new_w, new_h), Image.LANCZOS)
    # 轻度锐化提升清晰感
    return upscaled.filter(ImageFilter.UnsharpMask(radius=1, percent=120, threshold=3))


def resize_image(
    img: Image.Image,
    width: int | None = None,
    height: int | None = None,
) -> Image.Image:
    orig_w, orig_h = img.size
    if width and height:
        return img.resize((width, height), Image.LANCZOS)
    elif width:
        ratio = width / orig_w
        return img.resize((width, int(orig_h * ratio)), Image.LANCZOS)
    elif height:
        ratio = height / orig_h
        return img.resize((int(orig_w * ratio), height), Image.LANCZOS)
    return img
```

```python
# picedit/backend/services/birefnet_service.py
"""
BiRefNet-general 模型单例服务。
首次调用 get_model() 时从 HuggingFace 下载并缓存模型。
"""
from __future__ import annotations
import threading
import torch
from PIL import Image
from torchvision import transforms
from transformers import AutoModelForImageSegmentation

_model = None
_lock = threading.Lock()
_device = "cuda" if torch.cuda.is_available() else "cpu"

_transform = transforms.Compose([
    transforms.Resize((1024, 1024)),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])


def get_model():
    global _model
    if _model is None:
        with _lock:
            if _model is None:
                print("Loading BiRefNet-general model...")
                _model = AutoModelForImageSegmentation.from_pretrained(
                    "ZhengPeng7/BiRefNet-general",
                    trust_remote_code=True,
                )
                _model.to(_device)
                _model.eval()
                print(f"BiRefNet-general loaded on {_device}")
    return _model


def remove_background(image: Image.Image) -> Image.Image:
    """输入 RGB/RGBA 图像，返回带透明背景的 RGBA 图像。"""
    model = get_model()
    orig_size = image.size
    rgb_image = image.convert("RGB")

    input_tensor = _transform(rgb_image).unsqueeze(0).to(_device)

    with torch.no_grad():
        preds = model(input_tensor)[-1].sigmoid().cpu()

    # preds shape: (1, 1, H, W)，值 0~1，1 为前景
    mask_tensor = preds[0].squeeze(0)  # (H, W)
    mask_pil = transforms.ToPILImage()(mask_tensor).resize(orig_size, Image.LANCZOS)

    result = image.convert("RGBA")
    result.putalpha(mask_pil)
    return result
```

```python
from services.birefnet_service import remove_background

router = APIRouter(prefix="/api", tags=["remove-bg"])

    if not file.content_type.startswith("image/"):
```

```python
# picedit/backend/routers/resize.py
from fastapi import APIRouter, UploadFile, File, Form, HTTPException
from fastapi.responses import Response
from typing import Optional
from services.image_service import bytes_to_pil, resize_image, pil_to_png_bytes

router = APIRouter(prefix="/api", tags=["resize"])


@router.post("/resize")
async def resize(
    file: UploadFile = File(...),
    width: Optional[int] = Form(None, ge=1, le=8192),
    height: Optional[int] = Form(None, ge=1, le=8192),
):
    if not file.content_type.startswith("image/"):
        raise HTTPException(status_code=400, detail="Only image files are supported")

    if width is None and height is None:
        raise HTTPException(status_code=400, detail="Provide at least one of: width, height")

    data = await file.read()
    try:
        img = bytes_to_pil(data)
    except Exception:
        raise HTTPException(status_code=400, detail="Invalid image data")

    result = resize_image(img, width=width, height=height)
    return Response(content=pil_to_png_bytes(result), media_type="image/png")
```

```js
// picedit/frontend/src/api/client.js
const BASE = "http://localhost:8000/api";

async function postImage(endpoint, file, extraFields = {}) {
  const form = new FormData();
  form.append("file", file);
  for (const [key, value] of Object.entries(extraFields)) {
    form.append(key, String(value));
  }
  const res = await fetch(`${BASE}${endpoint}`, { method: "POST", body: form });
  if (!res.ok) {
    const err = await res.json().catch(() => ({ detail: "Unknown error" }));
    throw new Error(err.detail || `HTTP ${res.status}`);
  }
  return res.blob(); // 返回图片 Blob
}

export const removeBg = (file) => postImage("/remove-bg", file);

export const upscale = (file, scale = 2.0) =>
  postImage("/upscale", file, { scale });

export const resize = (file, { width, height }) => {
  const extra = {};
  if (width) extra.width = width;
  if (height) extra.height = height;
  return postImage("/resize", file, extra);
};
```

```js
// picedit/frontend/src/hooks/useProcessor.js
import { useState, useCallback } from "react";
import { removeBg, upscale, resize } from "../api/client";

export function useProcessor() {
  const [file, setFile] = useState(null);           // 原始 File 对象
  const [feature, setFeature] = useState("remove-bg"); // 当前选中功能
  const [params, setParams] = useState({});          // 各功能参数
  const [resultUrl, setResultUrl] = useState(null);  // 处理后图片 URL
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const process = useCallback(async () => {
    if (!file) return;
    setLoading(true);
    setError(null);
    setResultUrl(null);
    try {
      let blob;
      if (feature === "remove-bg") {
        blob = await removeBg(file);
      } else if (feature === "upscale") {
        blob = await upscale(file, params.scale ?? 2.0);
      } else if (feature === "resize") {
        blob = await resize(file, {
          width: params.width ? Number(params.width) : undefined,
          height: params.height ? Number(params.height) : undefined,
        });
      }
      setResultUrl(URL.createObjectURL(blob));
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, [file, feature, params]);

  return {
    file, setFile,
    feature, setFeature,
    params, setParams,
    resultUrl,
    loading,
    error,
    process,
  };
}
```

```jsx
import { useRef, useState } from "react";
import { UploadCloud } from "lucide-react";

export default function UploadZone({ file, onFileChange }) {
  const inputRef = useRef(null);
  const [dragging, setDragging] = useState(false);

  const handleFile = (f) => {
    if (f && f.type.startsWith("image/")) onFileChange(f);
  };

  const onDrop = (e) => {
    e.preventDefault();
    setDragging(false);
    handleFile(e.dataTransfer.files[0]);
  };

  const preview = file ? URL.createObjectURL(file) : null;

  return (
    <div
      onClick={() => inputRef.current.click()}
      onDragOver={(e) => { e.preventDefault(); setDragging(true); }}
      onDragLeave={() => setDragging(false)}
      onDrop={onDrop}
      className={`relative cursor-pointer rounded-xl border-2 border-dashed transition-colors
        flex flex-col items-center justify-center min-h-48 overflow-hidden
        ${dragging ? "border-indigo-500 bg-indigo-50" : "border-gray-300 bg-white hover:border-indigo-400 hover:bg-gray-50"}`}
    >
      <input
        ref={inputRef}
        type="file"
        accept="image/*"
        className="hidden"
        onChange={(e) => handleFile(e.target.files[0])}
      />
      {preview ? (
        <img
          src={preview}
          alt="preview"
          className="max-h-64 max-w-full object-contain rounded-lg"
        />
      ) : (
        <>
          <UploadCloud size={40} className="text-gray-400 mb-2" />
          <p className="text-sm text-gray-500 font-medium">点击或拖拽图片到此处</p>
          <p className="text-xs text-gray-400 mt-1">支持 PNG、JPG、WEBP</p>

}
```
